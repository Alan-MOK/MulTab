使用脚本打包

```bash
chmod +x package.sh && ./package.sh
```

打包完成后，生成的应用位于 `./build/Build/Products/Debug/MulTab.app`，你可以将其复制到 `/Applications` 目录下：

/realease/MulTab.app


# 1. 编译 Release 版本
xcodebuild -project MulTab.xcodeproj \
  -scheme MulTab \
  -configuration Release \
  -derivedDataPath ./build \
  clean build

# 2. 创建 DMG
create-dmg \
  --volname "MulTab Installer" \
  --window-pos 200 120 \
  --window-size 600 400 \
  --icon-size 100 \
  --icon "MulTab.app" 175 120 \
  --app-drop-link 425 120 \
  "./release/MulTab.dmg" \
  "./build/Build/Products/Release/MulTab.app"

  branch
